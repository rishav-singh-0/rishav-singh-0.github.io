{{- $destination := .Destination -}} {{- $text := .Text | safeHTML -}} {{- $isMalformed := and (strings.HasPrefix $destination "({<") (strings.Contains $destination "ref") -}} {{- if $isMalformed -}} {{- /* Extract the content from the malformed ref shortcode */ -}} {{- $inner := $destination | findRE `“([^”]+)”` | index 0 | default "" | replaceRE `[“”]` "" -}} {{- if $inner -}} {{- $inner = $inner | strings.TrimSuffix "/" -}} {{- /* Split path, section and alias */ -}} {{- $pathAndSection := $inner -}} {{- if strings.Contains $inner "|" -}} {{- $parts := split $inner "|" -}} {{- $pathAndSection = index $parts 0 -}} {{- end -}} {{- $path := $pathAndSection -}} {{- $section := "" -}} {{- if strings.Contains $pathAndSection "#" -}} {{- $parts := split $pathAndSection "#" -}} {{- $path = index $parts 0 -}} {{- $section = index $parts 1 -}} {{- end -}} {{- /* The python script creates incorrect paths in links. It uses the full obsidian path e.g. /posts/0-inbox/obsidian but the file is created at /posts/obsidian. We need to extract the last part of the path to find the page. */ -}} {{- $pathParts := split $path "/" -}} {{- $fileName := last 1 $pathParts | index 0 -}} {{- $correctPath := printf "posts/%s" $fileName -}} {{- /* Find the page using the corrected path */ -}} {{- $page := .Site.GetPage $correctPath -}} {{- if $page -}} {{- $url := $page.RelPermalink -}} {{- if $section -}} {{- $url = printf "%s#%s" $url (strings.Anchorize $section) -}} {{- end -}} {{ $text }} {{- else -}} {{- /* Fallback for broken links */ -}} {{ $text }} {{- end -}} {{- else -}} {{- /* Fallback for malformed links that we can't parse */ -}} {{ $text }} {{- end -}} {{- else -}} {{- /* Default link rendering for normal links */ -}} {{ .Text | safeHTML }} {{- end -}}
